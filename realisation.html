Configuration technique

Configuration réseau avec nmcli

# Configuration réseau avec nmcli (mode interactif)
nmcli connection edit

# Configuration des adresses IPv4
nmcli> goto ipv4.addresses
nmcli ipv4.addresses> set 10.31.101.66/24

# Configuration du serveur DNS
nmcli ipv4> goto dns
nmcli ipv4.dns> set 10.0.2.2

# Configuration de la passerelle
nmcli ipv4> goto gateway
nmcli ipv4.gateway> set 10.0.0.1

# Sauvegarde
nmcli> save
nmcli> quit

Configuration du proxy

# Configuration obligatoire pour toutes les machines
export http_proxy=http://172.20.100.242:3142
export https_proxy=http://172.20.100.242:3142

# Configuration permanente dans /etc/environment
http_proxy="http://172.20.100.242:3142"
https_proxy="http://172.20.100.242:3142"

Accès SSH en chaîne

# Connexions SSH depuis le poste principal
ssh anabea@tjener
ssh anabea@cab1
ssh anabea@cab2

Gestion des conteneurs

# Création d'un conteneur Debian
sudo debootstrap stable /var/lib/machines/cab1

# Démarrage du conteneur
sudo systemd-nspawn -D /var/lib/machines/cab1

# Création des utilisateurs
adduser anabea
adduser alibo
